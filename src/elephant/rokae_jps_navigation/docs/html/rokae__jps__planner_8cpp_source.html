291void JPSPlanner::collision_detection_client(std::vector<octomap::point3d> &waypoints, octomap::OcTree &tree)
292{
293 check_status_ = true;
294 const std::string COLLISION_DETECTION_CLIENT_STR = "/rokae_arm/robot_collision_detection_node";
295 ros::service::waitForService(COLLISION_DETECTION_CLIENT_STR);
296 collision_detection_client_ = nh_.serviceClient<rokae_jps_navigation::CheckCollision>(COLLISION_DETECTION_CLIENT_STR);
297
298 resamplePath(waypoints, tarPoints_pose_[currGoal_id_]);
299
300 int iteration = 0;
301 std::vector<std::vector<float>> joint_configs;
302 joint_configs.emplace_back(prev_joint_configs_);
303
304 for (const auto &pose_msg: tarPoints_pose_msg_)
305 {
306 rokae_jps_navigation::CheckCollision collision_detection_srv;
307 collision_detection_srv.request.path_pose = pose_msg;
308 collision_detection_srv.request.prev_joints.assign(prev_joint_configs_.begin(), prev_joint_configs_.end());
309
310 octomap::OcTreeKey currNodeKey = tree.coordToKey(waypoints[iteration]);
311
312 if (collision_detection_client_.call(collision_detection_srv))
313 {
314 if(collision_detection_srv.response.getSolution) {
315 if(collision_detection_srv.response.isCollide) {
316 printf("manipulator collision\n");
317 setCurrNodeObs(currNodeKey, tree);
318 planning_status_ = PlanningState::NEED_REPLANNING;
319 } else {
320 curr_joint_configs_.assign(collision_detection_srv.response.curr_joints.begin(), collision_detection_srv.response.curr_joints.end());
321 prev_joint_configs_.assign(curr_joint_configs_.begin(), curr_joint_configs_.end());
322 printf("Current pose safe. Get joint configs.\n");
323 joint_configs.push_back(curr_joint_configs_);
324 }
325 }
326 else {
327 printf("violated state\n");
328 setCurrNodeObs(currNodeKey, tree);
329 planning_status_ = PlanningState::NEED_REPLANNING;
330 }
331 }
332 }
333
334 if (planning_status_ != PlanningState::NEED_REPLANNING) {
335 joint_configs_ = joint_configs;
336 } else {
337 printf("Current path contains some obstacles or unreachable points.\n");
338 }

575bool JPSPlanner::plan(octomap::OcTree &tree, JPS::GraphSearch &planner)
576{
577 path_.clear();
578 optimal_path_.clear();
579
580 path_cost_ = planner.DPF_Plan(tree, region_set, dist_map, potential_weight_);
581
582 std::vector<octomap::OcTreeKey> keys = planner.getPathKeys();
583 if (keys.size() < 1) {
584 if(planner_verbose_) {
585 std::cout << ANSI_COLOR_RED "Cannot find a path in interatively compute progress. Abort!" ANSI_COLOR_RESET << std::endl;
586 return false;
587 }
588 }
589
590 std::vector<octomap::point3d> path_points = keysToCoords(keys, tree);
591 // Simplify the raw path
592 path_ = removeCornerPts(path_points, tree);
593 path_ = removeCornerPts(path_, tree);
594 path_ = removeLinePts(path_, tree);
595
596 return true;
597}

608 printf("search region radius: %d\n", search_radius_);
609 printf("max iteration time: %d\n", max_iteration);
610 }
611
612 optimal_path_ = raw_path_;
613 std::vector<octomap::OcTreeKey> optimal_keys = coordsToKeys(optimal_path_, tree);
614 double prev_cost = std::numeric_limits<double>::infinity();
615
616

619 setSearchRegion(optimal_keys, search_radius_, potential_radius_, tree);

630 } else

637 }
638 }

683 double mask_val = mask_gain_ * (dist_ratio * dist_ratio);

685 dist_map[new_key] = mask_val;
686 }

692void JPSPlanner::setSearchRegion(std::vector<octomap::OcTreeKey> &keys, const int search_radius, const int potential_radius, octomap::OcTree &tree)
693{
694 // if radius is negative or the search_radius is bigger than potential_radius, there will exist no tunnel
695 if (search_radius <= 0 || potential_radius <=0 || search_radius >= potential_radius) {
696 return;
697 }
698
699 // clear previous region_set
700 region_set.clear();
701 dist_path_keys_ = keys;
702
703 // get tunnel region neighbors'increment set
704 std::vector<std::vector<int>> SEARCH_REGION_NEIGHBORS_SET;
705 getDirSet(search_radius, SEARCH_REGION_NEIGHBORS_SET);
706
707 // get obstacle neighbors'increment set for distance potential field map building
708 std::vector<std::vector<int>> OBS_DPF_NEIGHBORS_SET;
709 getDirSet(potential_radius, OBS_DPF_NEIGHBORS_SET);
710
711 // store OcTreeKeys of search region in region_set
712 // If the planner finds a node is in search region and this node is occupied, we need to calculate the distance potential field value around this node.
713 for (auto &key: dist_path_keys_) {
714 for (auto &sd: SEARCH_REGION_NEIGHBORS_SET) {
715 octomap::OcTreeKey new_key = expand(key, sd);
716 octomap::point3d new_coord = keyToCoord_modified(new_key, tree);
717 // if the new_coord violate the edge limitation, planner will continue to process the next node and will not add this node to the region_set.
718 if (isOutside(new_coord)) {
719 continue;
720 }
721 // if the region_set has not contained the 'new_key', that means the 'new_key' is a search region node we need to consider.
722 auto find_result = region_set.find(new_key);
723 if (find_result == region_set.end()) {
724 region_set.insert(new_key);
725 octomap::OcTreeNode* newNodePtr = tree.search(new_key);
726 // if the new node is occupied, we need to do DPF process
727 if (isOccupied(newNodePtr)) {
728 dist_map[new_key] = mask_gain_; // current 'new_key' node is occupied, so we assigned 'mask_gain_' value to it.
729 for (auto &od: OBS_DPF_NEIGHBORS_SET) {
730 octomap::OcTreeKey new_newKey = expand(new_key, od);
731 octomap::point3d new_newCoord = keyToCoord_modified(new_newKey, tree);
732 // if the neighbor around the occupied node is outside of the whole map limitation, we abandon this node to process the next neighbor node
733 if (isOutside(new_newCoord)) {
734 continue;
735 }
736 double dist_ratio = 1.0 - (double)(std::abs(od[0]) + std::abs(od[1]) + std::abs(od[2])) / (3*potential_radius);
737 double mask_val = mask_gain_ * (dist_ratio * dist_ratio);
738 if (dist_map.find(new_newKey) == dist_map.end()) { // new in 'dist_map'
739 dist_map[new_newKey] = mask_val;
740 } else {
741 dist_map[new_newKey] = std::max(dist_map[new_newKey], mask_val);
742 }
743 }
744 }
745 }
746 }
747 }
748}

784std::vector<octomap::point3d> JPSPlanner::removeLinePts(const std::vector<octomap::point3d> &waypoints, octomap::OcTree &tree)
785{
786
787 if (waypoints.size() < 3) {
788 printf("[JPS]: No enough points for 'Line Points' filtering!\n");
789 return waypoints;
790 }
791
792 /* removing obsolete points */
793
794 ROS_INFO_STREAM("waypoints size:" << waypoints.size());
795
796 std::vector<octomap::point3d> optimized_path;
797
798 /* front(): this function returns a direct reference */
799 optimized_path.push_back(waypoints.front());
800 size_t k = 2; // 'front' and 'back'
801
802 while (k < waypoints.size())
803 {
804 if (!freeStraightPath(optimized_path.back(), waypoints[k], tree)) {
805 optimized_path.push_back(waypoints[k - 1]);
806 }
807
808 k++;
809 }
810 optimized_path.push_back(waypoints.back());
811
812 std::reverse(optimized_path.begin(), optimized_path.end());
813 // reverse order
814 return optimized_path;
815}

817std::vector<octomap::point3d> JPSPlanner::removeCornerPts(const std::vector<octomap::point3d> &waypoints, octomap::OcTree &tree)
818{
819 if (waypoints.size() < 2) {
820 printf("[JPS]: No enough points for 'Corner Points' filtering!\n");
821 return waypoints;
822 }
823
824 // cut zigzag segment
825 std::vector<octomap::point3d> optimized_path;
826 octomap::point3d point1 = waypoints[0];
827 octomap::point3d point2 = waypoints[1];
828 octomap::point3d prev_point = point1;
829
830 optimized_path.push_back(point1);
831
832 double cost1, cost2, cost3;
833
834 if (freeStraightPath(point1, point2, tree)) {
835 cost1 = (point1 - point2).norm();
836 } else {
837 cost1 = std::numeric_limits<double>::infinity();;
838 }
839
840 for (size_t i = 1; i < waypoints.size() - 1; ++i)
841 {
842 point1 = waypoints[i];
843 point2 = waypoints[i+1];
844
845 if (freeStraightPath(point1, point2, tree)) {
846 cost2 = (point1 - point2).norm();
847 } else {
848 cost2 = std::numeric_limits<double>::infinity();;
849 }
850
851 if (freeStraightPath(prev_point, point2, tree)) {
852 cost3 = (prev_point - point2).norm();
853 } else {
854 cost3 = std::numeric_limits<double>::infinity();;
855 }
856
857 if (cost3 < cost1 + cost2) {
858 cost1 = cost3;
859 } else {
860 optimized_path.push_back(waypoints[i]);
861 cost1 = (point1 - point2).norm();
862 prev_point = point1;
863 }
864
865 }
866 optimized_path.push_back(waypoints.back());
867 // reverse order
868 std::reverse(optimized_path.begin(), optimized_path.end());
869
870 return optimized_path;
871}
